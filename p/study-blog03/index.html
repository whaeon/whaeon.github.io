func (h *Hero) DeleteNode(id int) {
	tmp := h
	flag := false
	for {
		if tmp.Next == nil {
			break
		}
		if tmp.Next.ID == id {
			flag = true
			break
		}
		tmp = tmp.Next
	}
	if flag {
		tmp.Next = tmp.Next.Next
	} else {
		fmt.Println("no found!")
	}
}
